# F8 LLM Ops Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `llm_providers` + `llm_models` DB tables, admin CRUD endpoints, model promotion via `sync-llm-registry` BullMQ job, and wire persona model resolution so `LlmClientService` uses the persona's configured model instead of a hardcoded string.

**Architecture:** `LlmOpsModule` owns CRUD for providers/models. `Persona` gets optional `conversationModelId` + `scoringModelId` FKs. At session start, `ChatGateway` resolves the persona's conversation model name and stores it in `WsClient`. `ScoringService` resolves the persona's scoring model name before calling LLM. `POST /llm/models/:id/promote` sets `isDefault=true` and enqueues `sync-llm-registry` job which pushes to LiteLLM admin API.

**Tech Stack:** NestJS 11, Prisma, BullMQ, Zod DTOs, TypeScript strict, LiteLLM admin API.

// src/modules/llm-ops/llm-ops.service.ts
import { Injectable } from '@nestjs/common';
import { InjectQueue } from '@nestjs/bullmq';
import { Queue } from 'bullmq';
import { PrismaService } from '../../core/database/prisma.service';
import { NotFoundException } from '../../core/errors/domain.errors';
import type {
  CreateProviderDto,
  UpdateProviderDto,
  CreateModelDto,
  UpdateModelDto,
  ModelQueryDto,
} from './dto/llm-ops.dto';

export const SYNC_REGISTRY_QUEUE = 'sync-llm-registry';

@Injectable()
export class LlmOpsService {
  constructor(
    private readonly prisma: PrismaService,
    @InjectQueue(SYNC_REGISTRY_QUEUE) private readonly syncQueue: Queue,
  ) {}

  // ── Providers ──────────────────────────────────────────────────────────────

  listProviders() {
    return this.prisma.llmProvider.findMany({ orderBy: { priority: 'desc' } });
  }

  createProvider(dto: CreateProviderDto) {
    return this.prisma.llmProvider.create({ data: dto });
  }

  async updateProvider(id: number, dto: UpdateProviderDto) {
    await this.assertProvider(id);
    const data: Record<string, unknown> = {};
    if (dto.name !== undefined) data['name'] = dto.name;
    if (dto.type !== undefined) data['type'] = dto.type;
    if (dto.baseUrl !== undefined) data['baseUrl'] = dto.baseUrl;
    if (dto.credentialRef !== undefined) data['credentialRef'] = dto.credentialRef;
    if (dto.isEnabled !== undefined) data['isEnabled'] = dto.isEnabled;
    if (dto.priority !== undefined) data['priority'] = dto.priority;
    if ('monthlyBudgetUsd' in dto) data['monthlyBudgetUsd'] = dto.monthlyBudgetUsd ?? null;
    return this.prisma.llmProvider.update({ where: { id }, data });
  }

  async disableProvider(id: number) {
    await this.assertProvider(id);
    return this.prisma.llmProvider.update({ where: { id }, data: { isEnabled: false } });
  }

  // ── Models ─────────────────────────────────────────────────────────────────

  listModels(query: ModelQueryDto) {
    return this.prisma.llmModel.findMany({
      where: {
        ...(query.providerId !== undefined ? { providerId: query.providerId } : {}),
        ...(query.capability !== undefined ? { capabilities: { has: query.capability } } : {}),
      },
      include: { provider: { select: { id: true, name: true, type: true } } },
      orderBy: { name: 'asc' },
    });
  }

  createModel(dto: CreateModelDto) {
    return this.prisma.llmModel.create({ data: dto });
  }

  async updateModel(id: number, dto: UpdateModelDto) {
    await this.assertModel(id);
    const data: Record<string, unknown> = {};
    if (dto.name !== undefined) data['name'] = dto.name;
    if (dto.providerId !== undefined) data['providerId'] = dto.providerId;
    if (dto.capabilities !== undefined) data['capabilities'] = dto.capabilities;
    if (dto.contextWindowTokens !== undefined) data['contextWindowTokens'] = dto.contextWindowTokens;
    if (dto.inputPricePerMillion !== undefined) data['inputPricePerMillion'] = dto.inputPricePerMillion;
    if (dto.outputPricePerMillion !== undefined) data['outputPricePerMillion'] = dto.outputPricePerMillion;
    if (dto.isDefault !== undefined) data['isDefault'] = dto.isDefault;
    return this.prisma.llmModel.update({ where: { id }, data });
  }

  async promoteModel(id: number) {
    const model = await this.assertModel(id);
    await this.prisma.$transaction([
      this.prisma.llmModel.updateMany({ data: { isDefault: false } }),
      this.prisma.llmModel.update({ where: { id }, data: { isDefault: true } }),
    ]);
    await this.syncQueue.add('sync', { modelId: id, modelName: model.name });
    return { id, promoted: true };
  }

  async resolveModel(modelId: number | null | undefined, fallback: string): Promise<string> {
    if (!modelId) return fallback;
    const model = await this.prisma.llmModel.findUnique({ where: { id: modelId } });
    return model?.name ?? fallback;
  }

  async getDefaultModelName(): Promise<string> {
    const model = await this.prisma.llmModel.findFirst({ where: { isDefault: true } });
    return model?.name ?? 'gpt-4o-mini';
  }

  private async assertProvider(id: number) {
    const p = await this.prisma.llmProvider.findUnique({ where: { id } });
    if (!p) throw new NotFoundException('LlmProvider', id);
    return p;
  }

  private async assertModel(id: number) {
    const m = await this.prisma.llmModel.findUnique({ where: { id } });
    if (!m) throw new NotFoundException('LlmModel', id);
    return m;
  }
}
```

// src/modules/llm-ops/sync-registry.processor.ts
import { Processor, WorkerHost } from '@nestjs/bullmq';
import { Logger } from '@nestjs/common';
import { Job } from 'bullmq';
import { ConfigService } from '@nestjs/config';
import { PrismaService } from '../../core/database/prisma.service';
import { SYNC_REGISTRY_QUEUE } from './llm-ops.service';
import type { Env } from '../../core/config/env.schema';

interface SyncJobData {
  modelId: number;
  modelName: string;
}

@Processor(SYNC_REGISTRY_QUEUE)
export class SyncRegistryProcessor extends WorkerHost {
  private readonly logger = new Logger(SyncRegistryProcessor.name);

  constructor(
    private readonly prisma: PrismaService,
    private readonly config: ConfigService<Env, true>,
  ) {
    super();
  }

  override async process(job: Job<SyncJobData>): Promise<void> {
    const model = await this.prisma.llmModel.findUnique({
      where: { id: job.data.modelId },
      include: { provider: true },
    });

    if (!model) return;

    const baseUrl = this.config.get('LITELLM_BASE_URL', { infer: true });
    const apiKey = this.config.get('LITELLM_API_KEY', { infer: true });

    const body = {
      model_name: model.name,
      litellm_params: {
        model: model.name,
        ...(model.provider.baseUrl ? { api_base: model.provider.baseUrl } : {}),
      },
    };

    try {
      const resp = await fetch(`${baseUrl}/model/new`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${apiKey}`,
        },
        body: JSON.stringify(body),
      });

      if (!resp.ok) {
        const text = await resp.text();
        this.logger.error(`LiteLLM sync failed: ${resp.status} ${text}`);
      } else {
        this.logger.log(`Model "${model.name}" synced to LiteLLM`);
      }
    } catch (err) {
      this.logger.error({ err }, 'LiteLLM sync error');
    }
  }
}

// src/modules/llm-ops/llm-ops.controller.ts
import { Controller, Get, Post, Patch, Delete, Body, Param, Query, ParseIntPipe } from '@nestjs/common';
import { LlmOpsService } from './llm-ops.service';
import { Permissions } from '../../core/auth/decorators/permissions.decorator';
import { ZodValidationPipe } from '../../core/errors/zod-validation.pipe';
import {
  CreateProviderDtoSchema,
  UpdateProviderDtoSchema,
  CreateModelDtoSchema,
  UpdateModelDtoSchema,
  ModelQueryDtoSchema,
  type CreateProviderDto,
  type UpdateProviderDto,
  type CreateModelDto,
  type UpdateModelDto,
  type ModelQueryDto,
} from './dto/llm-ops.dto';

@Controller('llm')
export class LlmOpsController {
  constructor(private readonly llmOpsService: LlmOpsService) {}

  // ── Providers ──────────────────────────────────────────────────────────────

  @Get('providers')
  @Permissions('llmops:read')
  listProviders() {
    return this.llmOpsService.listProviders();
  }

  @Post('providers')
  @Permissions('llmops:write')
  createProvider(
    @Body(new ZodValidationPipe(CreateProviderDtoSchema)) dto: CreateProviderDto,
  ) {
    return this.llmOpsService.createProvider(dto);
  }

  @Patch('providers/:id')
  @Permissions('llmops:write')
  updateProvider(
    @Param('id', ParseIntPipe) id: number,
    @Body(new ZodValidationPipe(UpdateProviderDtoSchema)) dto: UpdateProviderDto,
  ) {
    return this.llmOpsService.updateProvider(id, dto);
  }

  @Delete('providers/:id')
  @Permissions('llmops:write')
  disableProvider(@Param('id', ParseIntPipe) id: number) {
    return this.llmOpsService.disableProvider(id);
  }

  // ── Models ─────────────────────────────────────────────────────────────────

  @Get('models')
  @Permissions('llmops:read')
  listModels(@Query(new ZodValidationPipe(ModelQueryDtoSchema)) query: ModelQueryDto) {
    return this.llmOpsService.listModels(query);
  }

  @Post('models')
  @Permissions('llmops:write')
  createModel(
    @Body(new ZodValidationPipe(CreateModelDtoSchema)) dto: CreateModelDto,
  ) {
    return this.llmOpsService.createModel(dto);
  }

  @Patch('models/:id')
  @Permissions('llmops:write')
  updateModel(
    @Param('id', ParseIntPipe) id: number,
    @Body(new ZodValidationPipe(UpdateModelDtoSchema)) dto: UpdateModelDto,
  ) {
    return this.llmOpsService.updateModel(id, dto);
  }

  @Post('models/:id/promote')
  @Permissions('llmops:write')
  promoteModel(@Param('id', ParseIntPipe) id: number) {
    return this.llmOpsService.promoteModel(id);
  }
}
```
